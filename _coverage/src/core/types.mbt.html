///|
/// 遅延評価可能な値。Val は即値、Thunk は遅延評価
pub(all) enum Lazy[T] {
  Val(T)
  Thunk(() -> T)
}

///|
/// Lazy[T] を評価して値を取得
pub fn[T] Lazy::resolve(self : Lazy[T]) -> T {
  match self {
    Val(v) => v
    Thunk(f) => f()
  }
}

/// 値の格納・状態管理を担う最小単位。cell + committed + default を内包する。
/// core 内部専用（pub にしない）。
///
/// default_val は Lazy[T] で保持する。Thunk を使えば reset のたびに新しいインスタンスを
/// 生成できるため、Array 等の mutable T でも参照共有問題が起きない。

///|
priv struct ValCell[T] {
  cell : Ref[T]
  committed : Ref[Bool]
  default_val : Lazy[T]
}

///|
fn[T] ValCell::new(default : Lazy[T]) -> ValCell[T] {
  {
    cell: { val: default.resolve() },
    committed: { val: false },
    default_val: default,
  }
}

///|
/// この ValCell に紐づいた操作インターフェースを返す。
fn[T] ValCell::accessor(self : ValCell[T]) -> Accessor[T] {
  let acc : Accessor[T] = {
    get: fn() { self.cell.val },
    set: fn(v) {
      self.cell.val = v
      self.committed.val = true
    },
    set_value: fn(v) { self.cell.val = v },
    set_commit: fn() { self.committed.val = true },
    reset: fn() {
      self.cell.val = self.default_val.resolve()
      self.committed.val = false
    },
  }
  // reset/clone はテスト API 向け。本体コードでは未使用だが whitebox テストで参照されるため残す。
  ignore(acc.reset)
  let _ : (ValCell[T]) -> ValCell[T] = ValCell::clone
  acc
}

/// default_val で新しい独立した ValCell を作る。現在の cell.val はコピーしない。
/// Lazy[T] により Thunk の場合は clone ごとに新しいインスタンスが生成される。
/// 本体コードでは未使用だが whitebox テストで検証されるため残す。

///|
fn[T] ValCell::clone(self : ValCell[T]) -> ValCell[T] {
  ValCell::new(self.default_val)
}

///|
/// Core primitive: creates a value cell with its accessor from an immediate value.
/// This is the foundational building block for all option combinators.
/// DR-061: value() is the only true value primitive.
fn[T] make_value_cell(initial~ : T) -> (ValCell[T], Accessor[T]) {
  let vc = ValCell::new(Val(initial))
  (vc, vc.accessor())
}

///|
/// Core primitive: creates a value cell with its accessor from a lazy value.
/// Used for mutable defaults (e.g. Array) where each reset needs a fresh instance.
/// DR-061: lazy variant of make_value_cell.
fn[T] make_value_cell_lazy(initial~ : Lazy[T]) -> (ValCell[T], Accessor[T]) {
  let vc = ValCell::new(initial)
  (vc, vc.accessor())
}

///|
/// Core primitive: creates a parallel accumulation cell.
/// DR-061 Phase 2: parallel(Opt[T]) -> Opt[Array[T]] の共通基盤。
/// append と rest が共有する「配列に要素を蓄積する」パターンを抽出。
///
/// Returns (valcell, accessor, push_fn, reset_fn) where:
/// - accessor is the raw Accessor for the underlying Array[T] cell
/// - push_fn(v) appends v to the array and marks committed
/// - reset_fn() resets to a fresh empty array and clears committed
fn[T] make_parallel_cell() -> (
  ValCell[Array[T]],
  Accessor[Array[T]],
  (T) -> Unit,
  () -> Unit,
) {
  let (vc, acc) = make_value_cell_lazy(initial=Thunk(fn() { [] }))
  let push = fn(v : T) {
    (acc.get)().push(v)
    (acc.set_commit)()
  }
  let reset = fn() { (acc.reset)() }
  (vc, acc, push, reset)
}

///|
/// ValCell への操作をクロージャ束として提供。Opt や post_hook から値を操作する統一 API。
/// core 内部専用。
///
/// - set(v)      : value と committed の両方を更新。set(v) = set_value(v) + set_commit()
/// - set_value(v): value のみセット。committed マークは変更しない
/// - set_commit(): committed マークのみセット。value は変更しない
/// - reset()     : value も committed もリセットする（value = default, committed = false）
priv struct Accessor[T] {
  get : () -> T
  set : (T) -> Unit
  set_value : (T) -> Unit
  set_commit : () -> Unit
  reset : () -> Unit
}
